115template<typename T>
116T reverseBits(T num, size_t nb = sizeof(T) * CHAR_BIT) {
117
118 if (nb > std::numeric_limits<T>::digits) {
119 throw std::invalid_argument("The number of reversal bits exceeds the number of bits in the number");
120 }
121
122 T rv = 0;
123
124 for (size_t i = 0; i < nb; ++i, num >>= 1) {
125 rv = (rv << 1) | (num & 0x01);
126 }
127
128 return rv;
129}
